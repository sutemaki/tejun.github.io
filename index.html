vbsで複数セル（行）の内容を1つの変数に格納したい

VBS（VBScript）単体では Excelの複数セルをそのまま1変数に入れることはできないので、
基本は ループで読み込んで文字列に連結する形になります。

例：A1～A5の内容を1つの変数にまとめる

Dim xl
Dim wb
Dim ws
Dim i
Dim result

Set xl = CreateObject("Excel.Application")
Set wb = xl.Workbooks.Open("C:\test.xlsx")
Set ws = wb.Worksheets(1)

result = ""

For i = 1 To 5
    result = result & ws.Cells(i,1).Value & vbCrLf
Next

MsgBox result

wb.Close False
xl.Quit
resultの中身
A1の内容
A2の内容
A3の内容
A4の内容
A5の内容
改行なしで1行にする場合
result = result & ws.Cells(i,1).Value & ","
結果

AAA,BBB,CCC,DDD
範囲指定で取得（少し高速）
Dim data

data = ws.Range("A1:A5").Value

For i = 1 To UBound(data,1)
    result = result & data(i,1) & vbCrLf
Next

できます。Excel VBA の UserForm + ListBox で実装するのが一番素直です。

読み込み処理

Sub LoadMailText()

    Dim txt As String
    Dim arr
    Dim i As Long
    
    txt = CreateObject("Scripting.FileSystemObject") _
        .OpenTextFile("C:\mail.txt").ReadAll
        
    arr = Split(txt, "-----Original Message-----")
    
    MailCount = UBound(arr)
    
    ReDim MailSender(MailCount)
    ReDim MailDate(MailCount)
    ReDim MailSubject(MailCount)
    ReDim MailBody(MailCount)
    
    For i = 0 To MailCount
    
        MailSender(i) = GetField(arr(i), "From:")
        MailDate(i) = GetField(arr(i), "Sent:")
        MailSubject(i) = GetField(arr(i), "Subject:")
        MailBody(i) = arr(i)
        
    Next i
    
End Sub
フィールド取得関数

Function GetField(txt As String, key As String) As String

    Dim lines
    Dim i As Long
    
    lines = Split(txt, vbCrLf)
    
    For i = 0 To UBound(lines)
    
        If InStr(lines(i), key) = 1 Then
            GetField = Replace(lines(i), key, "")
            Exit Function
        End If
        
    Next
    
End Function
④ UserForm 初期化
Private Sub UserForm_Initialize()

    Dim i As Long
    
    LoadMailText
    
    For i = 0 To MailCount
    
        lstMail.AddItem MailSender(i)
        lstMail.List(lstMail.ListCount - 1, 1) = MailDate(i)
        lstMail.List(lstMail.ListCount - 1, 2) = MailSubject(i)
        
    Next
    
End Sub
⑤ 選択時に本文表示
Private Sub lstMail_Change()

    Dim i As Long
    
    i = lstMail.ListIndex
    
    If i >= 0 Then
        txtBody.Value = MailBody(i)
    End If
    
End Sub
⑥ 貼り付け
Private Sub cmdPaste_Click()

    Dim i As Long
    
    i = lstMail.ListIndex
    
    If i >= 0 Then
    
        Range("A1").Value = MailSender(i)
        Range("B1").Value = MailDate(i)
        Range("C1").Value = MailSubject(i)
        Range("D1").Value = MailBody(i)
        
    End If
    
End Sub
