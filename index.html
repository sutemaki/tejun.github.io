========================= # 内容に応じてフォントサイズ調整 # ========================= # テキストボックスの中に文字が必ず収まるように、 # 表示文字列の長さとテキストボックスの大きさからフォントサイズを自動調整する function FitTxt($txt) { # 空の場合は標準サイズに戻して終了 if ([string]::IsNullOrWhiteSpace($txt)) { $out.Font = New-Object System.Drawing.Font("Meiryo UI",10) return } # 結果欄はボタンの下に配置 $outY = $mg + $btnH + $sp $out.Location = New-Object System.Drawing.Point($mg, $outY) $out.Size = New-Object System.Drawing.Size($cw, $outH) # サイズが変わったのでフォントも再調整する FitTxt $out.Text } # ========================= # 一時メッセージ表示 # ========================= # 「曜日を削除しました」などの案内を0.5秒だけ表示し、 # その後に最終結果へ切り替えるための共通処理 function TmpMsg($msg, $fin) { # 先にメッセージ表示 $out.Text = $msg FitTxt $out.Text $form.Refresh() [System.Windows.Forms.Application]::DoEvents() Start-Sleep -Milliseconds 500 # その後、最終結果へ切り替え $out.Text = $fin FitTxt $out.Text $form.Refresh() } # ========================= # 整形処理 # ========================= # クリップボードから取得した文字列を見て、 # どのパターンに当てはまるかを判定し、整形後の文字列と種別を返す function FmtTxt($txt) { # 前後の余分な空白や改行を削る $txt = $txt.Trim() # 現在年は「年が今と違う場合だけ先頭にyyを付ける」判定に使う $nowYear = (Get-Date).Year # 整形処理 # 1) 曜日削除 # 例:2026/04/07 (火) 21:37 → 2026/04/07 21:37 # # 実際には月日や時間が1桁でも受け付け、 # 出力時は 2026/04/07 09:05 のように二桁にそろえる # また、() と （） のどちらも中身ごと削除する if ($txt -match '^(\d{4})/(\d{1,2})/(\d{1,2})\s*[（(][^）)]*[）)]\s*(\d{1,2}):(\d{2})$') { $yy = [int]$matches[1] $mm = [int]$matches[2] $dd = [int]$matches[3] $hh = [int]$matches[4] $mi = [int]$matches[5] return @{ Result = ('{0:D4}/{1:D2}/{2:D2} {3:D2}:{4:D2}' -f $yy, $mm, $dd, $hh, $mi) Type = "wd_del" } } # 2) メアド削除 # 豊田太郎 # # Outlookの送信者欄などを想定し、 # 表示名だけを残してメールアドレス部分を削除する if ($txt -match '^(.*?)\s*[<〈][^>〉]+[>〉]\s*$') { return @{ Result = $matches[1].Trim() Type = "mail_del" } } # 2)メアド削除「<」以降全部削除 # if ($txt -match '[<〈]') { # return @{ # Result = ($txt -split '[<〈]')[0].Trim() # Type = "mail_del" # } # } # 3) cronID変換 # 例:202604071330_001 → 2026/04/07 13:30 / 202604071330_001 # # yyyymmddhhmm_xxx の並びを、人が読める日時 + 元のcronID にする if ($txt -match '^(\d{4})(\d{2})(\d{2})(\d{2})(\d{2})_(\d+)$') { return @{ Result = "$($matches[1])/$($matches[2])/$($matches[3]) $($matches[4]):$($matches[5]) / $txt" Type = "cron_add" } } # 4) cronID削除 # 例:2026/04/07 13:30 / 202604071330_001 → 2026/04/07 13:30 # # 「日時 / cronID」の形式から、先頭の日時だけを取り出す if ($txt -match '^(\d{4}/\d{2}/\d{2} \d{2}:\d{2})\s*/\s*\d{12}_\d+$') { return @{ Result = $matches[1] Type = "cron_cut" } } # 5) スケジュール登録の形に変換 # 例:2026/04/07 13:30 → 04.07.13.30 # 年が今年と違う場合は 27.04.07.13.30 のように # # ただし、これは単独の日時に対して常に動かすのではなく、 # 前回の処理が #4 の「cronID削除」だった場合のみ動かす if ( $script:lastFmt -eq "cron_cut" -and $txt -match '^(\d{4})/(\d{2})/(\d{2}) (\d{2}):(\d{2})$' ) { $yy = [int]$matches[1] if ($yy -eq $nowYear) { $res = "$($matches[2]).$($matches[3]).$($matches[4]).$($matches[5])" } else { $y2 = $matches[1].Substring(2,2) $res = "$y2.$($matches[2]).$($matches[3]).$($matches[4]).$($matches[5])" } return @{ Result = $res Type = "sch_fmt" } } # 補助変換: # 4/7 13:00 → 2026/04/07 13:00 # # Excelで日付として読ませやすいように、現在年を補完して # yyyy/MM/dd HH:mm 形式へそろえる if ($txt -match '^(\d{1,2})/(\d{1,2})\s+(\d{1,2}):(\d{2})$') { $mm = [int]$matches[1] $dd = [int]$matches[2] $hh = [int]$matches[3] $mi = [int]$matches[4] return @{ Result = ('{0:D4}/{1:D2}/{2:D2} {3:D2}:{4:D2}' -f $nowYear, $mm, $dd, $hh, $mi) Type = "date_add" } } return $null } # ========================= # クリップボード処理 # ========================= # クリップボードから文字列を取得し、 # 整形対象なら整形・再コピー、未対応ならそのまま表示する function DoClip { # 画像やファイルなど、テキスト以外は処理しない if (-not [System.Windows.Forms.Clipboard]::ContainsText()) { $out.Text = "クリップボードがテキストじゃありません" FitTxt $out.Text return } # ========================= # HSV→RGB変換 # ========================= # S3アラーム時の色変化用。 # 色相(H)を少しずつ進めて、なめらかに色が変わるようにする function Hsv2Rgb($h, $s, $v) { $c = $v * $s $x = $c * (1 - [math]::Abs((($h / 60.0) % 2) - 1)) $m = $v - $c if ($h -lt 60) { $r = $c; $g = $x; $b = 0 } elseif ($h -lt 120) { $r = $x; $g = $c; $b = 0 } elseif ($h -lt 180) { $r = 0; $g = $c; $b = $x } elseif ($h -lt 240) { $r = 0; $g = $x; $b = $c } elseif ($h -lt 300) { $r = $x; $g = 0; $b = $c } else { $r = $c; $g = 0; $b = $x } return [System.Drawing.Color]::FromArgb( [int](($r + $m) * 255), [int](($g + $m) * 255), [int](($b + $m) * 255) ) } #
