========================= # 内容に応じてフォントサイズ調整 # ========================= # 文字列の長さとテキストボックスの大きさからフォントサイズを自動調整 function Fittxt($txt) { # 空の場合は標準サイズに戻して終了 if ([string]::IsNullOrWhiteSpace($txt)) { $out.Font = New-Object System.Drawing.Font("Meiryo UI",10) return } # 一時タイトル表示 # ========================= # 指定したメッセージを、一定時間だけタイトルに表示 function Tmpttl($msg) { $script:tmpttl = $msg $script:tmpttlto = (Get-Date).AddMilliseconds(900) } # ========================= # Excel用変換用の日付解析 # ========================= # 可能な限り対応して DateTimeとして認識させる。 # 年ありは yyyy/MM/dd HH:mm # 年なしは MM/dd HH:mm function Convsubdate($txt) { $src = $txt.Trim() if ([string]::IsNullOrWhiteSpace($src)) { return $null } # 全角空白を半角へ $src = $src -replace '　', ' ' # 全角括弧を半角括弧へ $src = $src -replace '（', '(' $src = $src -replace '）', ')' # 曜日など括弧内表記を削除 $src = $src -replace '\s*\([^)]*\)\s*', ' ' # 半の処理。　例:「14時半」→「14時30分」 $src = $src -replace '(\d{1,2})時半', '$1時30分' # 和文の日付記号の変換 # 例: 2026年4月7日 14時30分 → 2026/4/7 14:30 $src = $src -replace '年', '/' $src = $src -replace '月', '/' $src = $src -replace '日', ' ' $src = $src -replace '時', ':' $src = $src -replace '分', '' # 複数空白を1つに $src = $src -replace '\s{2,}', ' ' $src = $src.Trim() # 「14時」が 前処理で「14:」になるので 14:00 に補完 $src = $src -replace '(\s\d{1,2}):$', '$1:00' $fmtsy = @( 'yyyy/M/d H:mm', 'yyyy/M/d HH:mm', 'yyyy/MM/dd H:mm', 'yyyy/MM/dd HH:mm' ) $fmtsn = @( 'M/d H:mm', 'MM/dd H:mm', 'M/d HH:mm', 'MM/dd HH:mm' ) $ci = [System.Globalization.CultureInfo]::InvariantCulture $dt = [datetime]::MinValue # 年ありで解釈 foreach ($f in $fmtsy) { if ([datetime]::TryParseExact($src, $f, $ci, [System.Globalization.DateTimeStyles]::None, [ref]$dt)) { return @{ Result = $dt.ToString('yyyy/MM/dd HH:mm') Type = 'sub_dt' } } } # 年なしで解釈 foreach ($f in $fmtsn) { if ([datetime]::TryParseExact($src, $f, $ci, [System.Globalization.DateTimeStyles]::None, [ref]$dt)) { return @{ Result = $dt.ToString('MM/dd HH:mm') Type = 'sub_dt' } } } return $null } # ========================= # 整形処理 # ========================= # クリップボードから取得した文字列を見て、 # どのパターンに当てはまるかを判定し、整形後の文字列と種別を返す function Fmttxt($txt) { # 前後の余分な空白や改行を削る $txt = $txt.Trim() # 現在年は「年が今と違う場合だけ先頭にyyを付ける」判定に使う $nowyear = (Get-Date).Year # 整形処理 # 1) 曜日削除 # 例:2026/04/07 (火) 21:37 → 2026/04/07 21:37 # # 実際には月日や時間が1桁でも受け付け、 # 出力時は 2026/04/07 09:05 のように二桁にそろえる # また、() と （） のどちらも中身ごと削除する if ($txt -match '^(\d{4})/(\d{1,2})/(\d{1,2})\s*[（(][^）)]*[）)]\s*(\d{1,2}):(\d{2})$') { $yy = [int]$matches[1] $mm = [int]$matches[2] $dd = [int]$matches[3] $hh = [int]$matches[4] $mi = [int]$matches[5] return @{ Result = ('{0:D4}/{1:D2}/{2:D2} {3:D2}:{4:D2}' -f $yy, $mm, $dd, $hh, $mi) Type = "wd_del" } } # 2) メアド削除 # 豊田太郎 # # Outlookの送信者欄などを表示名だけを残して # メールアドレス部分を削除する if ($txt -match '^(.*?)\s*<[^>]+>\s*$') { return @{ Result = $matches[1].Trim() Type = "mail_del" } } # 3) cronID変換 # 例:202604071330_001 → 2026/04/07　13:30　／　202604071330_001 # # yyyymmddhhmm_xxx の並びを、日時 + cronID に変換 if ($txt -match '^(\d{4})(\d{2})(\d{2})(\d{2})(\d{2})_(\d+)$') { return @{ Result = "$($matches[1])/$($matches[2])/$($matches[3])　$($matches[4]):$($matches[5])　／　$txt" Type = "cron_con" } } # 4) cronID削除 # 例:2026/04/07　13:30　／　202604071330_001 → 2026/04/07　13:30 # # 「日時 ／ cronID」の形式から、先頭の日時だけを取り出し、 # 日付と時刻の間を全角スペースにそろえる if ($txt -match '^(\d{4}/\d{2}/\d{2})[　 ](\d{2}:\d{2})[　 ]*[／/][　 ]*\d{12}_\d+$') { return @{ Result = "$($matches[1])　$($matches[2])" Type = "cron_cut" } } # 5) スケジュール登録の形に変換 # 例:2026/04/07　13:30 → 04.07.13.30 # 年が今年と違う場合は 27.04.07.13.30 のように # # 前回の処理が #4 の「cronID削除」だった場合のみ動かす if ( $script:lastfmt -eq "cron_cut" -and $txt -match '^(\d{4})/(\d{2})/(\d{2})[　 ](\d{2}):(\d{2})$' ) { $yy = [int]$matches[1] if ($yy -eq $nowyear) { $res = "$($matches[2]).$($matches[3]).$($matches[4]).$($matches[5])" } else { $y2 = $matches[1].Substring(2,2) $res = "$y2.$($matches[2]).$($matches[3]).$($matches[4]).$($matches[5])" } return @{ Result = $res Type = "sch_fmt" } } # Excel用日時変換 # 例:2026年4月7日 14時 → 2026/04/07 14:00 # 2026年4月7日(火) 14時30分 → 2026/04/07 14:30 # 4月7日(火) 13時30分 → 04/07 13:30 # 4/7 (火)13:30 → 04/07 13:30 # # 年ありは yyyy/MM/dd HH:mm # 年なしは MM/dd HH:mm $sub = Convsubdate $txt if ($null -ne $sub) { return $sub } return $null } # 結果の表示 $out.Text = $res Fittxt $res } # ========================= # HSV→RGB変換 # ========================= # S3アラーム時の色変化用。 # 色相(H)を少しずつ進めて色を変更する function Hsv2rgb($h, $s, $v) { $c = $v * $s $x = $c * (1 - [math]::Abs((($h / 60.0) % 2) - 1)) $m = $v - $c if ($h -lt 60) { $r = $c; $g = $x; $b = 0 } elseif ($h -lt 120) { $r = $x; $g = $c; $b = 0 } elseif ($h -lt 180) { $r = 0; $g = $c; $b = $x } elseif ($h -lt 240) { $r = 0; $g = $x; $b = $c } elseif ($h -lt 300) { $r = $x; $g = 0; $b = $c } else { $r = $c; $g = 0; $b = $x } return [System.Drawing.Color]::FromArgb( [int](($r + $m) * 255), [int](($g + $m) * 255), [int](($b + $m) * 255) ) } #
