function Convsubdate($txt) { $src = $txt.Trim() if ([string]::IsNullOrWhiteSpace($src)) { return $null } # 全角数字・全角記号をできるだけ半角へ寄せる $src = $src.Normalize([System.Text.NormalizationForm]::FormKC) # 曜日など括弧内表記を削除 $src = $src -replace '\s*\([^)]*\)\s*', ' ' # 「14時半」などを「14時30分」へ寄せる $src = $src -replace '(\d{1,2})時半', '$1時30分' # 和文の日付記号を一般的な形へ寄せる $src = $src -replace '年', '/' $src = $src -replace '月', '/' $src = $src -replace '日', ' ' $src = $src -replace '時', ':' $src = $src -replace '分', '' # 複数空白を1つに $src = $src -replace '\s{2,}', ' ' $src = $src.Trim() # 「14時」→ 前処理後は「14:」になるので 14:00 に補完 $src = $src -replace '(\s\d{1,2}):$', '$1:00' ... }
